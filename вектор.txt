#include <iostream>

#include <vector>
#include <algorithm>
#include <iterator>

using namespace std;

void display(char* str)
{
	cout << str << endl;
}

int main()
{
	cout << "First container" << endl;

	cout << "Max size: " << arr1.max_size() << endl;

	cout << "Capacity: " << arr1.capacity() << endl;

		arr1.push_back((char*)"19,8");

	cout << "Capacity: " << arr1.capacity() << endl;

	cout << "Ce: " << arr1.capacity() << endl;
	


	cout << endl;
	for_each(arr1.begin(), arr1.end(), display);
